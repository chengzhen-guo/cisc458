/* Procedure to generate random numbers */

fun randint (var result : like integer, modulus : like integer) is
    /* random number generation */
    if multiplier == 17 then
        multiplier = 19;
    else 
        multiplier = 17;
    end;
    seed = (seed * multiplier) % 10000;
    result = (seed % modulus) + 1;
end /*randint*/;
